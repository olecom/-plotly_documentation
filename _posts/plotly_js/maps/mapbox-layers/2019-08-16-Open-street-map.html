--- name: OpenStreetMap tiles, no token needed plot_url: https://codepen.io/plotly/embed/wvwgOrW/?height=510&theme-id=15263&default-tab=result language: plotly_js suite: mapbox-layers order: 3 sitemap: false arrangement: horizontal markdown_content: | Here is a simple map rendered with "open-street-map" tiles, without needing a Mapbox Access Token. --- Plotly.d3.csv( "https://raw.githubusercontent.com/plotly/datasets/master/2015_06_30_precipitation.csv", function(err, rows) { function unpack(rows, key) { return rows.map(function(row) { return row[key]; }); } var data = [ { type: "scattermapbox", text: unpack(rows, "Globvalue"), lon: unpack(rows, "Lon"), lat: unpack(rows, "Lat"), marker: { color: "fuchsia", size: 4 } } ]; var layout = { dragmode: "zoom", mapbox: { style: "open-street-map", center: { lat: 38, lon: -90 }, zoom: 3 }, margin: { r: 0, t: 0, b: 0, l: 0 } }; Plotly.newPlot("gd", data, layout); } );
